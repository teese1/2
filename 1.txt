private void LoginButton_Click(object sender, RoutedEventArgs e)
{
         
    var email = LoginEnter.Text.Trim();
    var passw = PasswordEnter.Password.Trim();

    if (string.IsNullOrEmpty(email) || string.IsNullOrEmpty(passw))
    {
        MessageBox.Show("Заполните все поля!");
        return; 
    }

    using var db = new ShoeshopContext  ();

    try
    {
        var user = db.Users
            .Include(u => u.Role)
            .FirstOrDefault(u => u.Email == email && u.Passw == passw);
        if (user == null)
        {
            MessageBox.Show("Неверный Email или пароль!");
            return;
        }

        CurrentUser.Id = user.IdUser;
        CurrentUser.FullName = $"{user.MiddleName} {user.FirstName} {user.LastName}";
        CurrentUser.Role = user.Role.RoleName;
        OpenProdWin();
    }
    catch (Exception ex)
    {
        MessageBox.Show($"Ошибка при подключении к базе данных: {ex.Message}");
    }
}

    Close();
}

ImagePathConverter
public class ImagePathConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        if (value == null)
        {
            return "/Images/picture.png";
        }

        string imageName = value.ToString().Trim();

        if (string.IsNullOrEmpty(imageName))
        {
            return "/Images/picture.png";
        }
        return $"/Images/{imageName}";
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        throw new NotImplementedException();
    }
}
SaleToBackgroundConverter
public class SaleToBackgroundConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        if (value != null && int.TryParse(value.ToString(), out int sale))
        {
            if (sale >= 15) return Brushes.LightGreen;
        }
        return Brushes.Transparent;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        throw new NotImplementedException();
    }
}
CrossedPriceConverter
public class CrossedPriceConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        if (value != null && int.TryParse(value.ToString(), out int sale))
        {
            if (sale > 0)
            {
                return TextDecorations.Strikethrough;
            }
        }
        return null;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        throw new NotImplementedException();
    }
}
FinalPriceConverter
public class FinalPriceConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        if (value is Product product)
        {
            decimal finalPrice = product.Price * (1 - (decimal)product.Sale / 100);
            return $"{finalPrice:N0} руб.";
        }
        return "0 руб.";
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        throw new NotImplementedException();
    }
}

public partial class CatalogWindow : Window
{
    public CatalogWindow()
    {
        InitializeComponent();
        UserNameTB.Text = CurrentUser.FullName;

        AdminPanel.Visibility = CurrentUser.Role == "Администратор" ? Visibility.Visible : Visibility.Collapsed;

        LoadSuppliers();
        LoadProdData();
    }

    private void LoadSuppliers()
    {
        try
        {
            using var db = new ShoeshopContext ();
            var suppliers = db.Suppliers.ToList();
            suppliers.Insert(0, new Supplier { IdSup = 0, SupName = "Все поставщики" });

            SupplierCB.ItemsSource = suppliers;
            SupplierCB.SelectedIndex = 0;
        }
        catch (Exception ex) { MessageBox.Show(ex.Message); }
    }

    private void LoadProdData()
    {
        if (ProductsLB == null || SupplierCB == null || SortCB == null || SearchTB == null) return;

        try
        {
            using var db = new ShoeshopContext ();
            var query = db.Products
                .Include(p => p.Cat)
                .Include(p => p.Manuf)
                .Include(p => p.Sup)
                .Include(p => p.ProdNameNavigation)
                .AsQueryable();

            if (!string.IsNullOrEmpty(SearchTB.Text))
            {
                string text = SearchTB.Text.ToLower();
                query = query.Where(p => p.Descrip.ToLower().Contains(text) ||
                                       p.Cat.CatName.ToLower().Contains(text));
            }

            if (SupplierCB.SelectedValue is int supId && supId > 0)
            {
                query = query.Where(p => p.SupId == supId);
            }

            if (SortCB.SelectedIndex == 1) query = query.OrderBy(p => p.Count);
            if (SortCB.SelectedIndex == 2) query = query.OrderByDescending(p => p.Count);

            ProductsLB.ItemsSource = query.ToList();
        }
        catch (Exception ex) { MessageBox.Show(ex.Message); }
    }

    private void SearchTB_TextChanged(object sender, TextChangedEventArgs e) => LoadProdData();
    private void SupplierCB_SelectionChanged(object sender, SelectionChangedEventArgs e) => LoadProdData();
    private void SortCB_SelectionChanged(object sender, SelectionChangedEventArgs e) => LoadProdData();

    private void ExitBtn_Click(object sender, RoutedEventArgs e)
    {
        new AuthorizationWindow ().Show();
        this.Close();
    }
}
